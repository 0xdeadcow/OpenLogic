\olsection{Axioms and Rules for Quantifiers}

\begin{defn}[Axioms for quantifiers] 
The \emph{axioms} governing quantifiers are
all instances of the following:
\begin{align}
\taq{Q1} & \lforall[x][!B] \lif !B(t); \\
\tag{Q2} & !B(t) \lif \lexists[x][!B].
\end{description} 
for any ground term~$t$.
\end{defn}

\begin{defn}[Rules for quantifiers]
\item If $!B \lif !A(a)$ already occurs and $a$ does not occur in $\Gamma$,
  then $!B \lif \lforall[x][!A(x)]$ is a correct inference step.
\item If $!A(a) \lif !B$ already occurs and $a$ does not occur in $\Gamma$,
  then $\lexists[x][!A(x)] \lif !B$ is a correct inference step.
\end{defn}
