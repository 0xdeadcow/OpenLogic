\olsection{Free Variables and Sentences}

\begin{defn}[Free occurrences of a !!{variable}]
The \emph{free} occurrences of a !!{variable} in a !!{formula} are defined
inductively:
\begin{enumerate}
\item If $!A$ is atomic, all !!{variable} occurrences in $!A$ are free.

\tagitem{prvNot}{If $!A$ is a !!{formula}, then the free !!{variable}
  occurrences of $\lnot !A$ are exactly those of $!A$.}{}

\tagitem{prvAnd}{If $!A$ and $!B$ are !!{formula}s, then the free
  !!{variable} occurrences in $(!A \land !B)$ are those in $!A$
  together with those in $!B$.}{}

\tagitem{prvOr}{If $!A$ and $!B$ are !!{formula}s, then the free
  !!{variable} occurrences in $(!A \lor !B)$ are those in $!A$
  together with those in $!B$.}

\tagitem{prvIf}{If $!A$ and $!B$ are !!{formula}s, then the free
  !!{variable} occurrences in $(!A \lif !B)$ are those in $!A$
  together with those in $!B$.}{}

\tagitem{prvIff}{If $!A$ and $!B$ are !!{formula}s, then the free
  !!{variable} occurrences in $(!A \liff !B)$ are those in $!A$
  together with those in $!B$.}{}

\tagitem{prvAll}{If $!A$ is a !!{formula}, then the free !!{variable}
  occurrences in $\lforall[x][!A]$ are all of those in~$!A$ except for
  occurrences of~$x$.}{}

\tagitem{prvEx}{If $!A$ is a !!{formula}, then the free !!{variable}
  occurrences in $\lexists[x][!A]$ are all of those in~$!A$ except for
  occurrences of~$x$.}{}
\end{enumerate}
\end{defn}

\begin{defn}[Sentence]
A !!{formula} $!A$ is a \emph{sentence} iff it contains no free
occurrences of !!{variable}s.
\end{defn}

\begin{defn} 
A term $t$ is \emph{free for $x$ in $!A$} if $x$ does not occur in
$!A$ within the scope of a quantifier $\lforall[z]$ binding a !!{variable}
$z$ occurring in~$t$.
\end{defn}
