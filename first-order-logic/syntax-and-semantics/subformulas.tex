\olsection{Subformulas}

\begin{explain}
It is often useful to talk about the !!{formula}s that ``make up'' a given
!!{formula}.  We call these the \emph{subformulas}.  Any !!{formula} counts as
a !!{subformula} of itself; a subformula of $!A$ other than $!A$ itself is
a \emph{proper !!{subformula}}.
\end{explain}

\begin{defn}[Sub-!!{formula}]
The set of \emph{sub-!!{formula}s} of a !!{formula}~$!A$,
$\SubFrm{!A}$, is defined inductively as follows:
\begin{enumerate}
\item If $!A$ is atomic, then $\SubFrm{!A} = \{!A\}$.

\tagitem{prvNot}{$\SubFrm{\lnot !A} = \{\lnot !A\} \cup \SubFrm{!A}$.}{}

\tagitem{prvAnd}{If $!B$ is also a !!{formula}, then $\SubFrm{!A \land
    !B} = \{!A \land !B\} \cup \SubFrm{!A} \cup \SubFrm{!B}$.}{}

\tagitem{prvOr}{If $!B$ is also a !!{formula}, then $\SubFrm{!A \lor
    !B} = \{!A \lor !B\} \cup \SubFrm{!A} \cup \SubFrm{!B}$.}{}

\tagitem{prvIf}{If $!B$ is also a !!{formula}, then $\SubFrm{!A \lif
    !B} = \{!A \lif !B\} \cup \SubFrm{!A} \cup \SubFrm{!B}$.}{}

\tagitem{prvIff}{If $!B$ is also a !!{formula}, then $\SubFrm{!A \liff
    !B} = \{!A \liff !B\} \cup \SubFrm{!A} \cup \SubFrm{!B}$.}{}

\tagitem{prvAll}{If $x$ is a !!{variable}, then
  $\SubFrm{\forall[x][!A]} = \{\lforall[x][!A]\} \cup \SubFrm{!A}$.}{}

\tagitem{prvEx}{If $x$ is a !!{variable}, then
  $\SubFrm{\lexists[x][!A]} = \{\lexists[x][!A]\} \cup
  \SubFrm{!A}$.}{}
\end{enumerate}
The set of all \emph{proper} sub-!!{formula}s of $!A$ is $\SubFrm{!A} \setminus
\{!A\}$. That is, a !!{formula} $!B$ is a proper !!{subformula} of $!A$ if and
only if $!B$ is a sub-!!{formula} of $!A$ but is distinct from $!A$.
\end{defn}

\begin{defn}[Immediate sub-!!{formula}]
Given !!{formula}s $!A$ and $!B$, define \emph{immediate sub-!!{formula}}
inductively as follows:
\begin{enumerate}
\item Atomic !!{formula}s have no immediate sub-!!{formula}s.

\tagitem{prvNot}{$!A$ is the immediate sub-!!{formula} of $\lnot
    !A$.}{} 

\tagitem{prvAnd}{$!A$ and $!B$ are the immediate
    sub-!!{formula}s of $(!A \land !B)$.}{}

\tagitem{prvOr}{$!A$ and $!B$ are the immediate sub-!!{formula}s of
  $(!A \lor !B)$.}{}

\tagitem{prvIf}{$!A$ and $!B$ are the immediate sub-!!{formula}s of
  $(!A \lif !B)$.}{}

\tagitem{prvIff}{$!A$ and $!B$ are the immediate sub-!!{formula}s of
  $(!A \liff !B)$.}{}

\tagitem{prvAll}{$!A$ is the immediate sub-!!{formula} of
  $\lforall[x][!A]$.}{}

\tagitem{prvEx}{$!A$ is the immediate sub-!!{formula} of
  $\lexists[x][!A]$.}{}
\end{enumerate}
\end{defn}

\begin{defn}[Sub-!!{formula}]
Define \emph{sub-!!{formula}} as follows:
\begin{enumerate}
\item If $!B$ is equal to $!A$, then $!B$ is a sub-!!{formula} of $!A$.
\item If $!B$ is an immediate sub-!!{formula} of $!A$, then $!B$ is a
  sub-!!{formula} of $!A$.
\item If $!C$ is a sub-!!{formula} of $!B$ and $!B$ is a sub-!!{formula} of
  $!A$, then $!C$ is also a sub-!!{formula} of $!A$.
\end{enumerate}
A !!{formula} $!B$ is a \emph{proper} sub-!!{formula} of $!A$ if and only if it
is a sub-!!{formula} of $!A$ and is not equal to $!A$.
\end{defn}

\begin{defn}[Proper sub-!!{formula}]
Define \emph{proper sub-!!{formula}} as follows:
\begin{enumerate}
\item If $!B$ is an immediate sub-!!{formula} of $!A$, then $!B$ is a
  proper sub-!!{formula} of $!A$.
\item If $!C$ is a proper sub-!!{formula} of $!B$ and $!B$ is a proper
  sub-!!{formula} of $!A$, then $!C$ is also a proper sub-!!{formula} of $!A$.
\end{enumerate}
\end{defn}

\begin{defn}[Sub-!!{formula}]
Define \emph{sub-!!{formula}} as follows:
\begin{enumerate}
\item If $!B$ is equal to $!A$, then $!B$ is a sub-!!{formula} of $!A$.
\item If $!B$ is a proper sub-!!{formula} of $!A$, then $!B$ is a
  sub-!!{formula} of~$!A$.
\end{enumerate}
\end{defn}
