\olsection{Soundness}

% Dana: define satisfaction & validity of sequents, prove soundness

\begin{thm}
If LK derives $\Gamma \Sequent \Delta$, then $\Gamma \Sequent \Delta$
is valid.
\end{thm}

\begin{thm}
\ollabel{soundness}
If $\Proves !A$ then $!A$ is valid.
\end{thm}

\begin{thm}
If $\Gamma \Proves !A$ then $\Gamma \Entails !A$.
\end{thm}

\begin{thm}
If $\Gamma$ is satisfiable, then it is consistent.
\end{thm}
